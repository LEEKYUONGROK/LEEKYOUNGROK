Point Point::operator / (Point tmp)
{
	if (this->x > tmp.x)
		tmp.x = (this->x / tmp.x);
	else if (this->x < tmp.x)
		tmp.x = (tmp.x / this->x);
	if (this->y > tmp.y)
		tmp.y = (this->y / tmp.y);
	else if (this->y < tmp.y)
		tmp.y = (tmp.y / this->y);
	return tmp;
}
